#!/bin/sh
exec tail -n +3 $0
# Custom GRUB menu entry for TechCorp Training VM

menuentry "TechCorp Training VM — Boot Standard" {
    set root=(hd0,msdos1)
    linux /boot/vmlinuz-$(uname -r) root=/dev/sda1
    initrd /boot/initrd.img-$(uname -r)
}
